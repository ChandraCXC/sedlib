231     /**
232      * Gets the enumerated utype associated with the specified keyword. Its
233      * possible the multiple keywords are associated with a single
234      * utype. Therefore an array of utypes are returned.
235      *
236      */
237     static public Integer[] getUtypeFromKeyword(String compName)
238     {
239         Integer utypeEnum[] = {INVALID_UTYPE};
240         if ((compName == null) || (compName.length () == 0))
241             return utypeEnum;
242 
243         // all keywords should be upper case
244         compName = compName.toUpperCase();
245 
246         utypeEnum = fitsMap.get(compName);
247 
248         if (utypeEnum == null)
249         {
250             // several keywords have variable numbers in them which 
251             // can be remapped as the letter n
252             if ((compName.matches("^COLLECT[1-9][0-9]*$")) ||
253                 (compName.matches("^CONTRIB[1-9][0-9]*$")) )
254 	    {
255 		//20110829: MCD fix for handling of these array keys.
256 		// if not array, keyword can be just a scalar key with no index
257                 compName = compName.replaceFirst("[1-9][0-9]*", "");
258                 utypeEnum = fitsMap.get(compName);
259 	    }
260 	    else if ((compName.matches ("^TDMIN[1-9][0-9]*$")) ||
261                 (compName.matches ("^TDMAX[1-9][0-9]*$")) ||
262                 (compName.matches ("^TCDLT[1-9][0-9]*$")) ||
263                 (compName.matches ("^TCTYP[1-9][0-9]*Z$")))
264             {
265                 compName = compName.replaceFirst ("[1-9][0-9]*", "n");
266                 utypeEnum = fitsMap.get (compName);
267             }
268 	    if (utypeEnum == null)
269             {
270                 utypeEnum = new Integer[1];
271                 utypeEnum[0] = INVALID_UTYPE;
272             }
273         }
274             
275         return utypeEnum;
276     }

307     /**
308      * Compare the string name of the utype with its enumerated counter part.
309      *
310      */
311      static public boolean compare (String utypeName, int utype)
312      {
313          boolean compareVal = false;
314          int utypeEnum = getUtypeFromString (utypeName);
315 
316          // first try the fits specific values
317          for (int ii=0; ii<fitsname.length; ii++)
318          {
319              if (utypeName.equals (fitsname[ii]) &&
320                  ((ii+max_enum) == utype))
321                  compareVal = true;
322          
323          }
324          
325          if (!compareVal)
326          {
327         	 utypeEnum = getUtypeFromString (utypeName);
328              compareVal = utypeEnum == utype;
329          }
330  
331          return compareVal;
332      }
