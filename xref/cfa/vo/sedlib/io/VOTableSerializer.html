146     /**
147      * Extract a VOTableObject from the Sed
148      */
149     protected VOTableObject processSed(Sed sed) throws SedInconsistentException
150     {
151         VOTableObject voTableObject = new VOTableObject ();
152 
153         VODocument voDocument = new VODocument();
154         VOElement root = (VOElement) voDocument.createElement(VOTableKeywords._VOTABLE);
155 
156 	// Check if SED is overriding the default namespace.
157 	String ns = sed.getNamespace();
158         if ( ns != null && !ns.equals( "" ) )
159         {
160 	    this.namespace = ns;
161         }
162 
163 	// Replace the trailing colon (it is suppossed to be there)
164 	// because it doesn't go in the 'xmlns' attribute.
165         String namespaceDeclaration=null;
166 	namespaceDeclaration = this.namespace.replaceFirst( ":$", "" );
167 	namespaceDeclaration = "xmlns:" + namespaceDeclaration;
168 
169         String[] nameArray = {"version",
170                                 "xmlns",
171                                 namespaceDeclaration,
172                                 "xmlns:xsi",
173                                 "xsi:noNamespaceSchemaLocation"};
174 
175         String[] valArray = {SedConstants.VOTABLE_VERSION,
176                                 IOConstants.IVOA_VOTABLE_SCHEMA,
177                                 IOConstants.IVOA_SPECTRUM_SCHEMA,
178                                 IOConstants.WWW_XMLSCHEMA_INSTANCE,
179                                 IOConstants.IVOA_VOTABLE_SCHEMA};
180 
181         // store off the root
182         voTableObject.root = root;
183 
184         // add header information
185         this.addAttributes(root, nameArray, valArray);
186         voDocument.appendChild(root);
187 
188         //RESOURCE element
189         VOElement resource = (VOElement) voDocument.createElement(VOTableKeywords._RESOURCE);
190 
191         //NOTE: what is correct utype to set to ?
192         resource.setAttribute(VOTableKeywords._UTYPE, VOTableKeywords.getName(VOTableKeywords.SED, this.namespace));
193         root.appendChild(resource);
194 
195         //On to the Data TABLE
196         if (sed.getNumberOfSegments () == 0)
197         	return voTableObject;
198 
199 
200         // Now add each segment in its own table.  
201         for (int ii=0; ii<sed.getNumberOfSegments (); ii++)
202         {
203             Segment segment = sed.getSegment (ii);
204         
205             TableElement voTable = null;
206             VOTablePointGroup pointsGroup = null;
207 
208             this.initializeSegment ();
209 
210             voTable = (TableElement) voDocument.createElement(VOTableKeywords._TABLE);
211             voTable.setAttribute(VOTableKeywords._UTYPE, VOTableKeywords.getName(VOTableKeywords.SEG, this.namespace));
212 
213             resource.appendChild(voTable);
214 
215             this.overrideParamInfo (segment);
216 
217             // add all the group information to the table
218             this.addSegmentToTable(segment, voTable);
219 
220             if (segment.isSetData ())
221             {
222                 pointsGroup = this.extractData (segment.getData (), voTable);
223 
224                 SedStarTable starTable = this.addDataToTable (pointsGroup, voTable);
225                 voTableObject.starTableList.add (starTable);
226 
227             }
228             else
229                 // add null spacer for segments without data
230                 voTableObject.starTableList.add (null);
231 
232         }
233 
234         return voTableObject;
235     }

264     /**
265      * Map an array of Param attribute to a VOElement Param. Multiple
266      * values will be space separated.
267      */
268     private void addSedParamArray(Param[] valueList,
269     		                     VOElement parent,  
270                                      int utype)
271     {
272         if (valueList.length == 0)
273             return;
274 
275         Param firstValue = valueList[0];
276         VOElement param = this.createParam (firstValue, parent, utype);
277         int arraySize = 0;
278        
279         if (firstValue.isSetValue() )
280         {
281             StringBuilder sbuf = new StringBuilder();
282             for ( int ii = 0; ii <  valueList.length; ii++ )
283             {
284                 if (valueList[ii] == null)
285                     break;
286                 if ( ii > 0)
287                      sbuf.append(" ");
288                 sbuf.append(valueList[ii].getValue());
289                 arraySize += 1;
290             }
291             param.setAttribute (VOTableKeywords._VALUE, sbuf.toString());
292         }
293 
294         if ((arraySize > 1) && !(firstValue instanceof TextParam))
295                 param.setAttribute(VOTableKeywords._ARRAYSIZE, Integer.toString (arraySize));
296 
297 
298         parent.appendChild (param);
299     }

301     /**
302      * Create a VOElement Param attribute and map a SED Param to it
303      */
304     private VOElement createParam (Param sedParam, 
305                                    VOElement parent,
306                                    int utype)
307     {
308         if (sedParam == null)
309             return null;
310 
311         Document document = parent.getOwnerDocument ();
312         VOElement param = (VOElement) document.createElement(VOTableKeywords._PARAM);
313         String utypeName = "";
314         String ucd = null;
315 
316         if (utype != VOTableKeywords.INVALID_UTYPE)
317         {
318             utypeName = VOTableKeywords.getName (utype, this.namespace);
319             param.setAttribute( VOTableKeywords._UTYPE, utypeName);
320         }
321 
322 
323         if (sedParam.isSetName ())
324             param.setAttribute(VOTableKeywords._NAME, sedParam.getName());
325         else
326         {
327             // if we don't have a name for this param, take the end
328             // of the utype and use it as the name.  We do this because the
329             // VOTable schema requires PARAMs to have a name.
330         	if (utype != VOTableKeywords.INVALID_UTYPE)
331         		param.setAttribute( VOTableKeywords._NAME, VOTableKeywords.getLastPartOfUtype( utypeName) );
332         }
333         
334         ucd = this.getParamUcd (sedParam, utype);
335         if (ucd != null)
336             param.setAttribute (VOTableKeywords._UCD, ucd);
337 
338 
339         if (sedParam.isSetValue() )
340             param.setAttribute (VOTableKeywords._VALUE, sedParam.getValue ());
341 
342 
343         if (sedParam instanceof DoubleParam)
344         {
345             DoubleParam dvalue = (DoubleParam)sedParam;
346             if (dvalue.isSetUnit ())
347                 param.setAttribute (VOTableKeywords._UNIT, dvalue.getUnit());
348             param.setAttribute(  VOTableKeywords._DATATYPE, "double" );
349         }
350         else if (sedParam instanceof IntParam)
351         {
352             IntParam ivalue = (IntParam)sedParam;
353             if (ivalue.isSetUnit ())
354                 param.setAttribute (VOTableKeywords._UNIT, ivalue.getUnit());
355             param.setAttribute(  VOTableKeywords._DATATYPE, "int" );
356         }
357         else if (sedParam instanceof TimeParam)
358         {
359             TimeParam tvalue = (TimeParam)sedParam;
360             if (tvalue.isSetUnit ())
361                 param.setAttribute (VOTableKeywords._UNIT, tvalue.getUnit());
362             param.setAttribute(  VOTableKeywords._DATATYPE, "char" );
363             param.setAttribute(VOTableKeywords._ARRAYSIZE, "*");
364         }
365         else
366         {
367             // this should cover TextParam, DateParam, SkyRegion
368             param.setAttribute(  VOTableKeywords._DATATYPE, "char" );
369             param.setAttribute(VOTableKeywords._ARRAYSIZE, "*");
370         }
371 
372         return param;
373     }

544     /**
545      * Add information from the CharacterizationAxis to the VOTable
546      */
547     private void addCharacterizationAxisToTable (CharacterizationAxis charAxis, VOElement parent, int utype) throws SedInconsistentException
548 
549     {
550         Document document = parent.getOwnerDocument ();
551         VOElement group;
552         int newUtype;
553         String functionName = "addCharacterizationAxisToTable";
554 
555         // add the name, ucd, and unit in a single param line
556         if (charAxis.isSetName () || charAxis.isSetUcd () || charAxis.isSetUnit ())
557         {
558             VOElement param = (VOElement) document.createElement(VOTableKeywords._PARAM);
559             String name;
560 
561             // use the name utype for this param
562             newUtype = this.mergeUtypes (utype, VOTableKeywords.SEG_CHAR_CHARAXIS_NAME, functionName);
563 
564            // force the name to be set to something
565            if (charAxis.isSetName() && !charAxis.getName().equals("") )
566            {
567               name = charAxis.getName();
568            }
569            else
570            {
571 	      // MCD NOTE: CharAxis.Name should default to that of DataAxis.Name and is an optional field.
572 	      //           while the UCD and Units are required... so serializing these as a single Param
573 	      //           may not be the best approach.
574               name = VOTableKeywords.getName(utype);
575               name = VOTableKeywords.getLastPartOfUtype(name);
576            }
577 
578            String values[] = {name, VOTableKeywords.getName(newUtype,this.namespace), charAxis.getUcd(), charAxis.getUnit(), "char", "*", name };
579 
580            if (!charAxis.isSetUcd ())
581                values[2] = VOTableKeywords.getUcd (utype);
582 
583 
584            this.addAttributes(param, paramAttributeNames, values);
585            parent.appendChild (param);  
586         }
587         
588         // add the rest
589         if (charAxis.isSetResolution ())
590         {
591             newUtype = this.mergeUtypes (utype, VOTableKeywords.SEG_CHAR_CHARAXIS_RESOLUTION, functionName);
592             this.addSedParam (charAxis.getResolution (), parent, newUtype);
593         }
594         if (charAxis.isSetCalibration ())
595         {
596             newUtype = this.mergeUtypes (utype, VOTableKeywords.SEG_CHAR_CHARAXIS_CAL, functionName);
597             this.addSedParam (charAxis.getCalibration (), parent, newUtype);
598 
599         }
600         if (charAxis.isSetCoordSystem ())
601         {
602             // no utype
603         }
604         if (charAxis.isSetCoverage ())
605         {
606             newUtype = this.mergeUtypes (utype, VOTableKeywords.SEG_CHAR_CHARAXIS_COV, functionName);
607             group = this.addGroup (newUtype, charAxis.getCoverage ().getGroupId (), parent);
608             this.addCoverageToTable (charAxis.getCoverage (), group, newUtype);
609             if (group.getChildren ().length > 0)
610                 parent.appendChild (group);
611         }
612         if (charAxis.isSetAccuracy ())
613         {
614             newUtype = this.mergeUtypes (utype, VOTableKeywords.SEG_CHAR_CHARAXIS_ACC, functionName);
615             group = this.addGroup (newUtype, charAxis.getAccuracy ().getGroupId (), parent);
616             this.addAccuracyToTable (charAxis.getAccuracy (), group, newUtype);
617             if (group.getChildren ().length > 0)
618                 parent.appendChild (group);
619         }
620         if (charAxis.isSetSamplingPrecision ())
621         {
622             newUtype = this.mergeUtypes (utype, VOTableKeywords.SEG_CHAR_CHARAXIS_SAMPPREC, functionName);
623             group = this.addGroup (newUtype, charAxis.getSamplingPrecision().getGroupId (), parent);
624             this.addSamplingPrecisionToTable (charAxis.getSamplingPrecision (), group, newUtype);
625             if (group.getChildren ().length > 0)
626                 parent.appendChild (group);
627         }
628 
629 
630         this.addCustomInfoToTable (charAxis, parent);
631 
632     }
