85      private void testMethod(String filename, SedFormat format, List<Oracle> oracleList)
86      {
87  
88          String inputFilename = SedLibTestUtils.mkInFileName( filename );
89          String outputFilename = SedLibTestUtils.mkOutFileName( filename );
90  
91  	/* Read input file */
92          Sed inputSed=null;
93          try {
94              inputSed = Sed.read(inputFilename, format);
95          } catch (SedParsingException ex) {
96              Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
97              fail("error while parsing the document: " +inputFilename);
98          } catch (SedInconsistentException ex) {
99              Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
100             fail("inconsistency detected in document: "+inputFilename);
101         } catch (IOException ex) {
102             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
103             fail("IO problems reading document: "+inputFilename);
104         }  catch (SedNoDataException ex) {
105             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
106             fail("No data found in segment: "+inputFilename);
107         }
108 
109 
110         /* Write file to disk */
111         try {
112             inputSed.write(outputFilename, format);
113         } catch (SedInconsistentException ex) {
114             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
115             fail("inconsistencies found in sed object. Can't serialize it to: " + outputFilename);
116         } catch (SedWritingException ex) {
117             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
118             fail("low level error while serializing sed object to " + outputFilename);
119         } catch (IOException ex) {
120             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
121             fail("I/O error while serializing sed object to " + outputFilename);
122         }
123 
124 
125         /* Read the new file */
126         Sed outputSed=null;
127         try {
128             outputSed = Sed.read(outputFilename, format);
129         } catch (SedParsingException ex) {
130             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
131             fail("error parsing file: " + outputFilename);
132         } catch (SedInconsistentException ex) {
133             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
134             fail("inconsistencies found in file: " + outputFilename);
135         } catch (IOException ex) {
136             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
137             fail("I/O error accessing" + outputFilename);
138         }  catch (SedNoDataException ex) {
139             Logger.getLogger(TestCase4_2.class.getName()).log(Level.SEVERE, null, ex);
140             fail("No data found in segment: "+inputFilename);
141         }
142 
143         checkSeds(inputSed, outputSed);
144 
145         for(int i=0; i<outputSed.getNumberOfSegments(); i++)
146         {
147             Segment s = outputSed.getSegment(i);
148             Oracle oracle = oracleList.get(i);
149             try {
150                 oracle.test(s);
151             } catch (Exception ex) {
152                 fail(ex.getMessage());
153             }
154         }
155 
156 	if ( !keep )
157 	    SedLibTestUtils.cleanupFiles( inputFilename );
158     }
